Enum ncollide::geom::geom::Geom

pub enum Geom<N, V, M> {
    PlaneGeom(Plane<N, V>),
    CompoundGeom(Arc<CompoundAABB<N, V, M, Geom<N, V, M>>>),
    ImplicitGeom(IGeom<N, V, M>),
}

Enumeration grouping all common shapes. Used to simplify collision detection dispatch.
